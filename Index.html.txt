<script>
  const screens = document.querySelectorAll(".screen");

  function showScreen(id) {
    screens.forEach((s) => {
      if (s.id === id) {
        s.classList.add("active");
      } else {
        s.classList.remove("active");
      }
    });
  }

  // Envelope & ribbon logic
  const envelope = document.getElementById("envelope");
  const ribbon = document.getElementById("ribbon");

  ribbon.addEventListener("click", () => {
    envelope.classList.add("open");
    setTimeout(() => {
      showScreen("screen-invite");
    }, 750);
  });

  // RSVP navigation
  function handleRsvpNext() {
    const nameInput = document.getElementById("guestName");
    const attendance = document.querySelector("input[name='attendance']:checked");

    if (!nameInput.value.trim()) {
      alert("Please enter your name.");
      return;
    }
    if (!attendance) {
      alert("Please let us know if you are joining us.");
      return;
    }

    if (attendance.value === "yes") {
      // Pass name into menu form
      document.getElementById("menuGuestName").value = nameInput.value.trim();
      showScreen("screen-menu");
    } else {
      // Not attending – show a simple confirmation
      const message = `
        <p>Dear <strong>${nameInput.value.trim()}</strong>,</p>
        <p>Thank you for letting us know. We are sorry you won’t be able to join us in Sorrento, but we truly appreciate your reply and send our love.</p>
        <p><strong>With love,</strong><br/>Yaaseen &amp; Sophia</p>
      `;
      document.getElementById("confirmationMessage").innerHTML = message;
      showScreen("screen-confirmation");
    }
  }

  // Menu form submission (currently just shows confirmation on-screen)
  const menuForm = document.getElementById("menuForm");
  menuForm.addEventListener("submit", function (e) {
    e.preventDefault();
